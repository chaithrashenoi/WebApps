use jadrn026;  drop table if exists vendor;
drop table if exists category;
drop table if exists product3;
 

create table vendor(
    id int AUTO_INCREMENT PRIMARY KEY,
    description varchar(20) NOT NULL );
    

create table category(
    id int AUTO_INCREMENT PRIMARY KEY,
    parent_id int,
    description varchar(20) NOT NULL );
    
create table product3(
    id int AUTO_INCREMENT PRIMARY KEY,
    sku char(7) NOT NULL,
    vendor_description varchar(20),
    category_description varchar(20),
    manufacturers_id varchar(50),
    description BLOB,
    features BLOB,
    cost varchar(10),
    retail varchar(10),
    product_image varchar(60) );

delete from vendor;
delete from category;
delete from product3;
